3   import java.util.Set;
4   
5   import javax.enterprise.context.ApplicationScoped;
6   import javax.enterprise.context.RequestScoped;
7   
8   import org.springframework.beans.factory.annotation.AnnotatedBeanDefinition;
9   import org.springframework.beans.factory.config.BeanDefinition;
10  import org.springframework.context.annotation.AnnotationScopeMetadataResolver;
11  import org.springframework.context.annotation.ScopeMetadata;
12  import org.springframework.context.annotation.ScopedProxyMode;
13  
14  public class CdiScopeMetadataResolver extends AnnotationScopeMetadataResolver {
15  
16      @Override
17      public ScopeMetadata resolveScopeMetadata(final BeanDefinition definition) {
18  
19          if (definition instanceof AnnotatedBeanDefinition) {
20              final AnnotatedBeanDefinition beanDefinition = (AnnotatedBeanDefinition) definition;
21              final ScopeMetadata metadata = new ScopeMetadata();
22              final Set<String> annotationTypes = beanDefinition.getMetadata().getAnnotationTypes();
23  
24              if (annotationTypes.contains(RequestScoped.class
25                  .getName())) {
26                  metadata.setScopeName("request");
27                  metadata.setScopedProxyMode(ScopedProxyMode.TARGET_CLASS);
28              } else if (annotationTypes
29                  .contains(ApplicationScoped.class.getName())) {
30                  metadata.setScopeName("singleton");
31              } else {
32                  return super.resolveScopeMetadata(definition);
33              }
34              return metadata;
35          } else {
36              return super.resolveScopeMetadata(definition);
37          }
38      }
39  }
