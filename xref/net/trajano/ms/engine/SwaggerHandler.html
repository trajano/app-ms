1   package net.trajano.ms.engine;
2   
3   import java.net.URI;
4   import java.util.Set;
5   
6   import javax.ws.rs.ApplicationPath;
7   import javax.ws.rs.core.Application;
8   import javax.ws.rs.core.MediaType;
9   
10  import org.reflections.Reflections;
11  
12  import com.fasterxml.jackson.core.JsonProcessingException;
13  
14  import io.swagger.annotations.Api;
15  import io.swagger.annotations.SwaggerDefinition;
16  import io.swagger.jaxrs.Reader;
17  import io.swagger.models.Swagger;
18  import io.vertx.core.Handler;
19  import io.vertx.core.buffer.Buffer;
20  import io.vertx.core.http.HttpHeaders;
21  import io.vertx.ext.web.Router;
22  import io.vertx.ext.web.RoutingContext;
23  import net.trajano.ms.engine.internal.swagger.ClonableSwagger;
24  
25  public class SwaggerHandler implements
26      Handler<RoutingContext>,
27      AutoCloseable {
28  
29      /**
30       * Convenience method to construct and register the routes to a Vert.x router
31       * with a base Spring application context.
32       *
33       * @param router
34       *            vert.x router
35       * @param applicationClasses
36       *            application classes
37       * @return the handlers
38       */
39      @SafeVarargs
40      public static SwaggerHandler[] multipleRegisterToRouter(final Router router,
41          final Class<? extends Application>... applicationClasses) {
42  
43          final SwaggerHandler[] ret = new SwaggerHandler[applicationClasses.length];
44          int i = 0;
45          for (final Class<? extends Application> applicationClass : applicationClasses) {
46              final SwaggerHandler requestHandler = new SwaggerHandler(applicationClass);
47              router.route(requestHandler.baseUriRoute())
48                  .useNormalisedPath(true)
49                  .handler(requestHandler);
50              ret[i++] = requestHandler;
51          }
52          return ret;
53  
54      }
55  
56      /**
57       * Convenience method to construct and register a single application route to a
58       * Vert.x router.
59       *
60       * @param router
61       *            vert.x router
62       * @param applicationClass
63       *            application class
64       * @return the handler
65       */
66      public static SwaggerHandler registerToRouter(final Router router,
67          final Class<? extends Application> applicationClass) {
68  
69          return multipleRegisterToRouter(router, applicationClass)[0];
70      }
71  
72      private final URI baseUri;
73  
74      private final ClonableSwagger swagger;
75  
76      public SwaggerHandler(
77          final Class<? extends Application> applicationClass) {
78  
79          final ApplicationPath annotation = applicationClass.getAnnotation(ApplicationPath.class);
80          if (annotation != null) {
81              baseUri = URI.create(annotation.value()).normalize();
82          } else {
83              baseUri = URI.create("/");
84          }
85  
86          Application application;
87          try {
88              application = applicationClass.newInstance();
89          } catch (InstantiationException
90              | IllegalAccessException e) {
91              throw new ExceptionInInitializerError(e);
92          }
93  
94          swagger = new ClonableSwagger();
95          final Reader swaggerReader = new Reader(swagger);
96          final Set<Class<?>> resourceClasses = application.getClasses();
97          if (resourceClasses.isEmpty()) {
98              final String packageName = applicationClass.getPackage().getName();
99              final Reflections reflections = new Reflections(packageName);
100             reflections.getTypesAnnotatedWith(Api.class).forEach(swaggerReader::read);
101             reflections.getTypesAnnotatedWith(SwaggerDefinition.class).forEach(swaggerReader::read);
102         } else {
103             swaggerReader.read(applicationClass);
104             resourceClasses.forEach(swaggerReader::read);
105         }
106 
107     }
108 
109     /**
110      * The base URI set in {@link ApplicationPath} annotation.
111      *
112      * @return the base URI
113      */
114     public URI baseUri() {
115 
116         return baseUri;
117     }
118 
119     /**
120      * The route path to the base URI. Basically base URI + "/*".
121      *
122      * @return The route path to the base URI.
123      */
124     public String baseUriRoute() {
125 
126         return baseUri().getPath();
127     }
128 
129     @Override
130     public void close() {
131 
132         // does nothing.
133     }
134 
135     public Swagger getSwagger() {
136 
137         return swagger;
138     }
139 
140     @Override
141     public void handle(final RoutingContext context) {
142 
143         try {
144             context.response().putHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON)
145                 .end(Buffer.buffer(io.swagger.util.Json.mapper()
146                     .writeValueAsBytes(swagger.withRoutingContext(context))));
147         } catch (final JsonProcessingException e) {
148             context.fail(e);
149         }
150     }
151 
152 }
