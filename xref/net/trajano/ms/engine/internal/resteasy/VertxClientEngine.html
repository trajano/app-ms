6   import java.security.NoSuchAlgorithmException;
7   
8   import javax.net.ssl.HostnameVerifier;
9   import javax.net.ssl.HttpsURLConnection;
10  import javax.net.ssl.SSLContext;
11  
12  import org.jboss.resteasy.client.jaxrs.ClientHttpEngine;
13  import org.jboss.resteasy.client.jaxrs.internal.ClientInvocation;
14  import org.jboss.resteasy.client.jaxrs.internal.ClientResponse;

16  import org.slf4j.LoggerFactory;
17  
18  import io.vertx.core.buffer.Buffer;
19  import io.vertx.core.http.HttpClient;
20  import io.vertx.core.http.HttpClientRequest;
21  import io.vertx.core.http.HttpMethod;
22  import io.vertx.core.http.RequestOptions;
23  import net.trajano.ms.engine.internal.Conversions;
24  import net.trajano.ms.engine.internal.VertxOutputStream;
25  
26  public class VertxClientEngine implements
27      ClientHttpEngine {
28  
29      private static final Logger LOG = LoggerFactory.getLogger(VertxClientEngine.class);
30  
31      private final HostnameVerifier hostnameVerifier;
32  
33      private final HttpClient httpClient;
34  
35      private final SSLContext sslContext;
36  
37      public VertxClientEngine(final HttpClient httpClient) {
38  
39          try {
40              this.httpClient = httpClient;
41              sslContext = SSLContext.getDefault();
42              hostnameVerifier = HttpsURLConnection.getDefaultHostnameVerifier();
43          } catch (final NoSuchAlgorithmException e) {
44              throw new ExceptionInInitializerError(e);
45          }
46      }
47  
48      @Override
49      public void close() {
50  
51          LOG.trace("closing {}", this);
52  
53      }
54  
55      @Override
56      public HostnameVerifier getHostnameVerifier() {
57  
58          return hostnameVerifier;
59      }
60  
61      @Override
62      public SSLContext getSslContext() {
63  
64          return sslContext;
65      }
66  
67      @Override
68      public ClientResponse invoke(final ClientInvocation request) {
69  
70          LOG.debug("{} {}", request.getMethod(), request.getUri());
71          final RequestOptions options = Conversions.toRequestOptions(request.getUri());
72          final HttpClientRequest httpClientRequest = httpClient.request(HttpMethod.valueOf(request.getMethod()), options);
73  
74          final VertxClientResponse clientResponse = new VertxClientResponse(request.getClientConfiguration(), httpClientRequest);
75  
76          LOG.debug("clientResponse={}", clientResponse);
77          request.getHeaders().asMap().forEach(httpClientRequest::putHeader);
78          httpClientRequest.setChunked(request.isChunked());
79  
80          if (request.isChunked()) {
81  
82              try (final VertxOutputStream os = new VertxOutputStream(httpClientRequest)) {
83                  request.writeRequestBody(os);
84                  LOG.trace("request body written on {}", this);
85                  return clientResponse;
86              } catch (final IOException e) {
87                  throw new UncheckedIOException(e);
88              } finally {
89                  httpClientRequest.end();
90              }
91          } else {
92              try (final ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
93                  request.writeRequestBody(baos);
94                  httpClientRequest.end(Buffer.buffer(baos.toByteArray()));
95                  return clientResponse;
96              } catch (final IOException e) {
97                  throw new UncheckedIOException(e);
98              }
99  
100         }
101 
102     }
